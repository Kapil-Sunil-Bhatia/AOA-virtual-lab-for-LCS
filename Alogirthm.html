ALGORITHM :

LCS-LENGTH(X, Y)
m = length[ X ]
n = length[ Y ]
for i = 1 to m
         do c[ i , 0 ] = 0
for j = 1 to n
         do c[ 0 , j ] = 0
for i = 1 to m
         do for j = 1 to n
              do if Xi == Yj
                then c[ i , j ] = c[ i – 1 , j – 1 ] + 1
                        b[i , j] = ARROW_CORNER
               else if c[ i – 1 , j ] > = c[ i , j – 1 ]
                        then c[ i , j ] = c[ i – 1 , j ]
                                b[ i , j ] = ARROW_UP
                        else c[ i , j ] = c[ i , j – 1 ]
                                b[ i , j ] = ARROW_LEFT
return c and b
PRINT-LCS ( b , X , i , j )
if i = 0 or j = 0
         then return
if b[ i , j ] == ARROW_CORNER
         then PRINT-LCS ( b , X , i – 1 , j – 1 )
              print Xi
else if b [ i , j ] == ARROW_UP
         then PRINT-LCS ( b , X , i – 1 , j )
         else PRINT-LCS( b , X , i , j – 1 )
